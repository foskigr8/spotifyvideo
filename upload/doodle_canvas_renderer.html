The renderer class — drop this into your project
class DoodleRenderer { constructor(canvas) { this.ctx = canvas.getContext('2d'); this.queue = []; this.busy = false; this.speed = 1; // multiply all durations by 1/speed this._applyPaperTexture(); } // Main entry — pass AI-returned command array renderCommands(cmds) { this.queue.push(...cmds); if (!this.busy) this._next(); } async _next() { if (!this.queue.length) { this.busy = false; return; } this.busy = true; const cmd = this.queue.shift(); await this._execute(cmd); this._next(); } async _execute(cmd) { switch (cmd.type) { case 'stroke': return await this._drawStroke(cmd); case 'text': return await this._drawText(cmd); case 'circle': return await this._drawCircle(cmd); case 'wait': return await this._wait(cmd.ms); case 'clear': return this._clear(); } } // Animate a path point-by-point (hand-drawn feel) async _drawStroke({points, color='#222', width=2.5, duration=600}) { const ctx = this.ctx; const ms = duration / this.speed; const delay = ms / points.length; ctx.strokeStyle = color; ctx.lineWidth = width; ctx.lineCap = 'round'; ctx.lineJoin = 'round'; for (let i = 1; i < points.length; i++) { ctx.beginPath(); ctx.moveTo(points[i-1].x, points[i-1].y); ctx.lineTo(points[i].x, points[i].y); ctx.stroke(); await this._wait(delay); } } // Typewriter-style hand font text async _drawText({text, x, y, size=28, color='#222', duration=500}) { const ctx = this.ctx; const delay = (duration / this.speed) / text.length; ctx.fillStyle = color; ctx.font = `700 ${size}px 'Caveat', cursive`; let drawn = ''; for (const ch of text) { drawn += ch; ctx.clearRect(x - 2, y - size, ctx.measureText(drawn).width + 40, size + 8); ctx.fillText(drawn, x, y); await this._wait(delay); } } _wait(ms) { return new Promise(r => setTimeout(r, ms)); } }
